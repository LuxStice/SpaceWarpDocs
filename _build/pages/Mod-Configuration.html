using SpaceWarp.API.Configuration;
using Newtonsoft.Json;
namespace Namespace;
[JsonObject(MemberSerialization.OptOut)]
[ModConfig]
public class Config {
   ...
}

Поля конфигурации¶

Чтобы создать новое поле, сделайте следующее (это создаст поле ввода/переключения). поле в редакторе конфигов в Space Warp)

[ConfigField("name")]
[ConfigDefaultValue(3)] // Optional
public int name;
